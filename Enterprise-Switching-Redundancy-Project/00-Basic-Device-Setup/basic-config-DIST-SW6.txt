! ===== BASIC CONFIG – DIST-SW6 (Distribution Porto) =====
hostname DIST-SW6

no ip domain-lookup
ip domain-name porto-gaia.lab

enable secret ENABLE_PASSWORD
username admin privilege 15 secret ADMIN_PASSWORD

! Link L3 com R5
interface Ethernet0/0
 ip address 10.51.0.6 255.255.255.0
 no shutdown

! Interface física para router-on-a-stick (subinterfaces vão noutro projeto)
interface Ethernet1/0
 no ip address
 no shutdown

crypto key generate rsa modulus 2048
ip ssh version 2

banner motd ^C
ACESSO RESTRITO
DIST-SW6 – Distribution Porto.
^C

line con 0
 logging synchronous
 exec-timeout 0 0
 privilege level 15

line vty 0 4
 logging synchronous
 exec-timeout 10 0
 login local
 transport input ssh
