! ===== BASIC CONFIG – R7 (Core Gaia) =====
hostname R7

no ip domain-lookup
ip domain-name porto-gaia.lab

enable secret ENABLE_PASSWORD
username admin privilege 15 secret ADMIN_PASSWORD

! Ligações ao R5 (backbone)
interface Ethernet0/2
 ip address 10.55.0.7 255.255.255.0
 no shutdown

interface Ethernet0/3
 ip address 10.77.0.7 255.255.255.0
 no shutdown

! (As subinterfaces .73/.74/.75 vão noutro projeto – router-on-a-stick/OSPF)

crypto key generate rsa modulus 2048
ip ssh version 2

banner motd ^C
ACESSO RESTRITO
Equipamento R7 – Core Gaia.
^C

line con 0
 logging synchronous
 exec-timeout 0 0
 privilege level 15

line vty 0 4
 logging synchronous
 exec-timeout 10 0
 login local
 transport input ssh
